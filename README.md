### Deployment

    npm install --save-dev gh-pages

    package.json add
        "homepage": "https://rnfsoft.github.io/crud-app-in-react-with-hooks"

        "scripts": {
            // ...
            "predeploy": "npm run build",
            "deploy": "gh-pages -d build"
        }

    npm run build

    npm run deploy
